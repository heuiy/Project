Sub ExportToPDF()
   Dim acadDoc As AcadDocument
   Set acadDoc = ThisDrawing
​
   ' PDF 출력 설정
   Dim PrinterName As String
   PrinterName = "DWG To PDF.pc3" ' PDF 프린터 이름
​
   Dim CustomPaperSize As String
   CustomPaperSize = "ISO_A4_(210.00_x_297.00_MM)" ' 종이 크기
​
   ' 도면 영역 설정
   Dim LowerLeft() As Double
   LowerLeft = Array(0, 0, 0) ' 하단 왼쪽 좌표

   Dim UpperRight() As Double
   UpperRight = Array(297, 210, 0) ' 상단 오른쪽 좌표
​
   ' PDF로 출력
   acadDoc.SetVariable "TileMode", 0
   acadDoc.ExportToPDF PrinterName, CustomPaperSize, LowerLeft, UpperRight, acExportPDFAll

End Sub
